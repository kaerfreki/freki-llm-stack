def parse_duration(s: str) -> int:
    """Parses duration strings like "2h15m", "45s", "1d2h30m5s" into a total number of seconds.
    Supported units: d, h, m, s. Units may be omitted but must appear in
    descending order at most once each; the string must contain at least one
    component.

    Args:
        s: The duration string to parse.

    Returns:
        The total number of seconds represented by the duration string.

    Raises:
        ValueError: If the input string is invalid (empty, unknown unit, wrong order, repeated unit, bare numbers).

    Examples:
        parse_duration("2h15m") == 13500
        parse_duration("45s") == 45
        parse_duration("1d2h30m5s") == 96605
    """
    if not s:
        raise ValueError("Input string cannot be empty.")

    pattern = re.compile(r"(\d+)?d?(\d+)h?(\d+)m?(\d+)s?")
    match = pattern.match(s)

    if not match:
        raise ValueError("Invalid duration format.")

    days = int(match.group(1) or 0)
    hours = int(match.group(2) or 0)
    minutes = int(match.group(3) or 0)
    seconds = int(match.group(4) or 0)

    if days > 0 and hours > 0:
      raise ValueError("Units must appear in descending order.")
    if minutes > 0 and hours == 0 and days > 0:
      raise ValueError("Units must appear in descending order.")
    if seconds > 0 and minutes == 0 and hours == 0 and days > 0:
        raise ValueError("Units must appear in descending order.")

    if days < 0 or hours < 0 or minutes < 0 or seconds < 0:
       raise ValueError("Duration values cannot be negative.")
   
    total_seconds = (days * 86400) + (hours * 3600) + (minutes * 60) + seconds

    return total_seconds
